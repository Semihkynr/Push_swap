void print_stack(t_stack *stack)
{
    t_stack *current = stack;

    while (current != NULL)
    {
        printf("Content: %d, Index: %d", current->content, current->index);
        if (current->next != NULL)
            printf("\n");
        current = current->next;
    }
    printf("\n");
}
